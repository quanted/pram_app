\documentclass[fleqn, oneside, 11pt]{article}%allows pdfs to be cropped and fleqn left justifies

\usepackage{amsmath}%sets math environment
\usepackage[active, tightpage]{preview}%crops pdf to border of equations
\DeclareMathSizes{10}{1}{1}{1}
\setlength\PreviewBorder{1pt}%sets size of border

\begin{document}
\begin{preview}
\begin{align*}%sets up equation environment so you can specify alignment with ampersands
f_{p} & = f_{p, max}(\frac{t}{T_{m}-T_{e}}) \;\;\;\;\;\;\;\;\; T_{e}<t<T_{m} \nonumber \\ 
f_{p} & = f_{p, max} \;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\; T_{m}<t<T_{r} \\ 
f_{p} & = 0 \;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\; T_{r}<t<T_{e} \nonumber \\
\text{where} \nonumber \\
f_{p} & =  \text{the fractional area of coverage at time t} \nonumber \\
f_{p, max} & =  \text{the maximum fractional area of plant coverage} \nonumber \\
T_{e} & =  \text{time of emergence} \nonumber \\
T_{m} & =  \text{time of maximum coverage} \nonumber \\
T_{r} & =  \text{time of removal} \nonumber \\
\end{align*} 
\end{preview}
